CREATE TABLE USUARIO(
    	USUID INT PRIMARY KEY IDENTITY(1,1) NOT NULL,
	USUCPF VARCHAR (14) NOT NULL,
	USUNOME VARCHAR(50) NOT NULL,
	USUEMAIL VARCHAR(50) NOT NULL,
	USUSENHA VARCHAR(20) NOT NULL,
	USUCELULAR VARCHAR(11) NULL,
	USUTIPO [int] NOT NULL
)

CREATE TABLE RESTAURANTE(
	RESID INT PRIMARY KEY IDENTITY(1,1) NOT NULL,
	RESNOME VARCHAR(30) NULL,
	RESDESCRICAO VARCHAR(max) NULL,
	RESENDERECEO VARCHAR(50) NULL,
	RESTELEFONE VARCHAR(11) NULL,
	RESHRFUNCINI VARCHAR(5) NULL,
	RESHRFUNCFIM VARCHAR(5) NULL,
	RESDIASFUNC VARCHAR(50) NULL
)
CREATE TABLE PEDIDO(
	PEDID INT PRIMARY KEY IDENTITY(1,1) NOT NULL,
	PEDSTATUS INT NOT NULL,
	PEDHRPEDIDO DATETIME NOT NULL,
	PEDCODMESA INT NOT NULL,
	PEDCPFCLI VARCHAR(14) NULL
)


CREATE TABLE MESA(
	MESCODMESA INT PRIMARY KEY IDENTITY(1,1) NOT NULL,
	MESSTATUS INT NOT NULL
)

CREATE TABLE CARDAPIO(
	CARID INT PRIMARY KEY IDENTITY(1,1) NOT NULL		
)

CREATE TABLE ITEMCARDAPIO(
	ITCID INT PRIMARY KEY IDENTITY(1,1) NOT NULL,
	ITCNOME VARCHAR(30) NULL,
	ITCDESCR VARCHAR(70) NULL,
	ITCPRECO FLOAT NOT NULL,	
)

--TABELA DE RELACIONAMENTO ENTRE O CARDAPIO E OS ITENS
CREATE TABLE RELITENSCRDP
(
	RLICIDITEM INT NOT NULL,
	RLICIDCARDAPIO INT NOT NULL
)

--TABELA DE RELACIONAMENTO ENTRE O PEDIDO E OS ITENS DO CARDAPIO
CREATE TABLE RELITENSPED
(
    RIPDIDPEDIDO INT NOT NULL,
	RIPIDITEM INT NOT NULL
)
